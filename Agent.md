### `brain/agent.py` — Real ADK agent definition
```python
# brain/agent.py
import os, importlib, pkgutil
from dotenv import load_dotenv

from google.adk.agents   import Agent
from google.adk.runners  import Runner
from google.adk.sessions import InMemorySessionService

from brain.model_router          import OpenRouterModel
from brain.tools.web_search      import search_web
from brain.tools.web_automation  import browse
from brain.tools.pc_control      import (move_mouse, click, type_text,
                                          press_key, screenshot, launch)
from brain.tools.file_system     import read_file, write_file, list_dir
from brain.tools.scheduler_ipc   import schedule_task

load_dotenv()

# ── Root ADK Agent ───────────────────────────────────────────────
root_agent = Agent(
    name        = "polyglot_agent",
    model       = OpenRouterModel(),       # OpenRouter wrapper (see §5)
    description = "General-purpose AI agent with PC control, "
                  "web browsing, scheduling, and plugin support.",
    instruction = (
        "You are a powerful AI assistant. You can control the user's PC, "
        "browse the web, search for information, manage files, and schedule "
        "tasks. Think step-by-step. Use tools whenever possible. "
        "Be concise and precise."
    ),
    tools = [
        search_web, browse,
        move_mouse, click, type_text, press_key, screenshot, launch,
        read_file, write_file, list_dir,
        schedule_task,
    ],
)

# ── ADK Session Memory (short-term, in-process) ──────────────────
_session_service = InMemorySessionService()

runner = Runner(
    agent           = root_agent,
    app_name        = "polyglot_agent",
    session_service = _session_service,
)

# ── Auto-load Skills (ADK sub-agents) ────────────────────────────
for _, name, _ in pkgutil.iter_modules(["brain/skills"]):
    mod = importlib.import_module(f"brain.skills.{name}")
    if hasattr(mod, f"{name}_agent"):
        root_agent.tools.append(getattr(mod, f"{name}_agent"))

# ── Auto-load Plugins (ADK callback hooks) ───────────────────────
for _, name, _ in pkgutil.iter_modules(["brain/plugins"]):
    mod = importlib.import_module(f"brain.plugins.{name}")
    if hasattr(mod, "before_model_call"):
        root_agent.before_model_callback = mod.before_model_call
    if hasattr(mod, "after_model_call"):
        root_agent.after_model_callback  = mod.after_model_call
```

### How ADK Session Memory Works

```
User Input
    │
    ▼
runner.run_async(session_id, user_id, message)
    │
    ├── ADK retrieves session from InMemorySessionService
    │       └── session.history  ← full turn-by-turn context (automatic)
    │
    ├── Agent reasons over history + calls @tool functions
    │
    └── ADK stores new turns back into session automatically
            └── persists for the lifetime of the process
```

- **`InMemorySessionService`** — built into `google-adk`, zero config, lives in RAM. Sessions are keyed by `(app_name, user_id, session_id)`.
- Each `runner.run_async()` call automatically reads and writes the session — **no manual memory management needed**.
- Swap out for persistence: `InMemorySessionService` → `DatabaseSessionService(db_url="sqlite:///sessions.db")`

```python
# brain/tools/pc_control.py
import ctypes, json
from google.adk.tools import tool

_lib = ctypes.CDLL("./native/build/libagent_native.so")

@tool
def move_mouse(x: int, y: int) -> str:
    """Move the mouse cursor to screen coordinates (x, y)."""
    _lib.move_mouse(x, y)
    return f"Mouse moved to ({x}, {y})"

@tool
def click(button: int = 0) -> str:
    """Click the mouse. button: 0=left, 1=right, 2=middle."""
    _lib.click_mouse(button)
    return "Clicked"

@tool
def type_text(text: str) -> str:
    """Type text via keyboard simulation."""
    _lib.type_text(text.encode())
    return f"Typed: {text}"

@tool
def press_key(combo: str) -> str:
    """Press a key combination e.g. 'ctrl+c', 'enter', 'alt+f4'."""
    _lib.press_key(combo.encode())
    return f"Pressed: {combo}"

@tool
def screenshot(path: str = "/tmp/screenshot.png") -> str:
    """Take a full-screen screenshot and save it to path."""
    _lib.take_screenshot(path.encode())
    return f"Screenshot saved to {path}"

@tool
def launch(cmd: str) -> str:
    """Launch an application or shell command."""
    _lib.launch_process(cmd.encode())
    return f"Launched: {cmd}"
```

## 4. 🧩 Short-Term Memory — Google ADK

The agent uses **`google-adk`'s `InMemorySessionService`** as its short-term memory. ADK handles full context accumulation automatically per session — no custom ring buffer needed.

### Chat Function Called by Rust (via PyO3)
```python
# brain/agent.py — append this helper
import asyncio
from google.genai import types as genai_types

async def chat(user_id: str, session_id: str, message: str) -> str:
    """Called by the Rust TUI engine for each user turn."""
    # Swap model based on task type before running
    root_agent.model = root_agent.model.select_for_task(message)

    content = genai_types.Content(
        role  = "user",
        parts = [genai_types.Part(text=message)],
    )
    response = ""
    async for event in runner.run_async(
        user_id     = user_id,
        session_id  = session_id,
        new_message = content,
    ):
        if event.is_final_response():
            response = event.response.parts[0].text
    return response

def chat_sync(user_id: str, session_id: str, message: str) -> str:
    """Synchronous wrapper for PyO3 / Rust calls."""
    return asyncio.run(chat(user_id, session_id, message))
```

## 5. 🔀 OpenRouter — Free Model + Large Context

### Custom ADK Model Wrapper (`brain/model_router.py`)

```python
# brain/model_router.py
import httpx, os
from google.adk.models.base_llm import BaseLlm

FREE_LARGE_CTX = [
    "google/gemini-2.0-flash-exp:free",        # 1M context — default
    "meta-llama/llama-3.3-70b-instruct:free",  # 128k context
    "deepseek/deepseek-r1:free",               # 164k context
]

class OpenRouterModel(BaseLlm):
    """Drop-in ADK model that routes all LLM calls through OpenRouter."""

    def __init__(self, model: str = FREE_LARGE_CTX[0]):
        self.model   = model
        self.api_key = os.getenv("OPENROUTER_API_KEY")
        self.base    = "https://openrouter.ai/api/v1"

    @property
    def model_name(self) -> str:
        return self.model

    def _call(self, messages: list, **kwargs) -> str:
        resp = httpx.post(
            f"{self.base}/chat/completions",
            headers={"Authorization": f"Bearer {self.api_key}"},
            json={"model": self.model, "messages": messages, **kwargs},
            timeout=120,
        )
        resp.raise_for_status()
        return resp.json()["choices"][0]["message"]["content"]

    def select_for_task(self, task: str) -> "OpenRouterModel":
        """Return the right model for the detected task type."""
        t = task.lower()
        if any(w in t for w in ["draw", "generate image", "picture", "flux"]):
            return OpenRouterModel("black-forest-labs/flux-1.1-pro")
        if any(w in t for w in ["video", "animate", "render video", "runway"]):
            return OpenRouterModel("runway/gen3a_turbo")
        if any(w in t for w in ["look at", "describe image", "what's in"]):
            return OpenRouterModel("google/gemini-2.0-flash-exp:free")
        return OpenRouterModel(FREE_LARGE_CTX[0])   # default: 1M ctx, free
```

### ADK Scheduler IPC Tool (`brain/tools/scheduler_ipc.py`)
```python
import socket, json
from google.adk.tools import tool

@tool
def schedule_task(cron: str, action: str, args: dict = {}) -> str:
    """
    Schedule a recurring task.
    cron: standard cron expression e.g. '0 9 * * 1'
    action: dotted skill name e.g. 'brain.skills.briefing.run'
    """
    payload = json.dumps({"cron": cron, "action": action, "args": args})
    with socket.socket(socket.AF_UNIX, socket.SOCK_STREAM) as s:
        s.connect("/tmp/agent_scheduler.sock")
        s.sendall(payload.encode())
    return f"Scheduled '{action}' with cron '{cron}'"
```

### Writing a Plugin (ADK Callback Hook)
```python
# brain/plugins/logger/__init__.py
from google.adk.agents.callback_context import CallbackContext
from google.adk.models                  import LlmRequest, LlmResponse

PLUGIN_META = {
    "name":    "logger",
    "hooks":   ["before_model_call", "after_model_call"],
    "version": "1.0.0",
}

def before_model_call(ctx: CallbackContext, req: LlmRequest):
    """Fires before every LLM call — log, modify, or short-circuit."""
    print(f"[plugin:logger] → {req.messages[-1]['content'][:80]}")

def after_model_call(ctx: CallbackContext, resp: LlmResponse):
    """Fires after every LLM response — log or transform output."""
    print(f"[plugin:logger] ← {str(resp)[:80]}")
```

## 11. 🖥️ Offline Models via Ollama ⚠️ Hardware Warning

> **Not recommended on weak hardware.** Even small models (3B–7B) need 4–8 GB RAM and benefit greatly from a GPU.  
> This section exists so others with capable machines can plug in offline models. The system degrades gracefully — if Ollama is unreachable, it falls back to OpenRouter automatically.

### How it fits in `model_router.py`

```python
# brain/model_router.py — append to OpenRouterModel

import httpx

OLLAMA_BASE = "http://localhost:11434"   # default Ollama port

OLLAMA_MODELS = {
    "small":  "phi3:mini",               # 3.8B — lightest usable model
    "medium": "mistral:7b-instruct",     # 7B — decent quality
    "large":  "llama3:8b-instruct",      # 8B — good quality, needs ~8GB RAM
    "code":   "codellama:7b",            # code tasks
    "vision": "llava:7b",               # image understanding offline
}

class OllamaModel(BaseLlm):
    """Offline model via Ollama. Falls back to OpenRouter on failure."""

    def __init__(self, model: str = OLLAMA_MODELS["small"]):
        self.model    = model
        self.base     = OLLAMA_BASE
        self._fallback = OpenRouterModel()   # automatic fallback

    @property
    def model_name(self) -> str:
        return f"ollama/{self.model}"

    def _call(self, messages: list, **kwargs) -> str:
        try:
            resp = httpx.post(
                f"{self.base}/api/chat",
                json={"model": self.model, "messages": messages, "stream": False},
                timeout=120,
            )
            resp.raise_for_status()
            return resp.json()["message"]["content"]
        except Exception as e:
            print(f"[ollama] offline or error ({e}) — falling back to OpenRouter")
            return self._fallback._call(messages, **kwargs)
```

## 12. 🧠 Live Neural Network — Multi-Agent Debate Visualiser

When a complex problem is detected, the agent spins up a **multi-agent debate loop** and renders a **live node graph in the TUI** showing agents arguing until consensus.

### Detection heuristic (`brain/agent.py`)
```python
COMPLEX_TRIGGERS = [
    "plan", "architect", "design", "solve", "debug", "compare",
    "analyse", "strategy", "how do i", "best way to", "trade-off",
]

def is_complex(message: str) -> bool:
    return any(w in message.lower() for w in COMPLEX_TRIGGERS) \
        or len(message.split()) > 40
```

### Multi-Agent Debate (`brain/debate.py`)
```python
# brain/debate.py
import asyncio
from google.adk.agents import Agent
from brain.model_router import OpenRouterModel

AGENT_ROLES = [
    ("Proposer",  "You propose a clear solution. Be bold."),
    ("Critic",    "You find flaws in the proposal. Be harsh but fair."),
    ("Mediator",  "You synthesise the debate into the best final answer."),
]

async def debate(problem: str, rounds: int = 3) -> tuple[str, list[dict]]:
    """
    Run a multi-agent debate and return (final_answer, debate_log).
    debate_log is streamed live to the TUI neural graph.
    """
    agents = [
        Agent(name=role, model=OpenRouterModel(),
              instruction=instruction, tools=[])
        for role, instruction in AGENT_ROLES
    ]
    history, log = [], []
    for round_n in range(rounds):
        for agent in agents:
            prompt   = f"Problem: {problem}\n\nDebate so far:\n" + \
                       "\n".join(f"{e['agent']}: {e['text']}" for e in history)
            response = await agent.model._call(
                [{"role": "user", "content": prompt}]
            )
            entry = {"round": round_n, "agent": agent.name,
                     "text": response, "agrees": _check_consensus(response)}
            history.append(entry)
            log.append(entry)
            yield entry                  # ← stream to TUI node graph
    # Mediator's last response is the final answer
    return history[-1]["text"], log

def _check_consensus(text: str) -> bool:
    agree_words = ["agree", "correct", "yes", "confirmed", "consensus"]
    return any(w in text.lower() for w in agree_words)
```

## 13. 🔗 AI SDK + models.dev Integration (OpenCode-style)

> OpenCode uses `@ai-sdk/openai` style providers from Vercel's AI SDK, then resolves model strings via `models.dev` registry. We mirror that pattern in Python using `httpx` + a provider registry dict — no JS dependency needed.

### How OpenCode does it (reference)
```typescript
// opencode uses this pattern internally:
import { createOpenAI } from "@ai-sdk/openai"
import { models }       from "models.dev"          // model registry

const provider = createOpenAI({ apiKey, baseURL })
const model    = provider(models["gpt-4o"].id)
```

### Our Python equivalent (`brain/providers/registry.py`)
```python
# brain/providers/registry.py
# Mirrors models.dev registry — hardcoded snapshot + live fetch fallback

import httpx, os

MODELS_DEV_URL = "https://models.dev/api/models.json"   # public registry

# Cached snapshot (works offline, mirrors models.dev format)
REGISTRY: dict[str, dict] = {
    # OpenRouter
    "gemini-2.0-flash-exp:free": {
        "id":        "google/gemini-2.0-flash-exp:free",
        "provider":  "openrouter",
        "context":   1_000_000,
        "vision":    True,
        "free":      True,
    },
    "deepseek-r1:free": {
        "id":        "deepseek/deepseek-r1:free",
        "provider":  "openrouter",
        "context":   164_000,
        "free":      True,
    },
    # DashScope (Alibaba)
    "qwen-max": {
        "id":        "qwen-max",
        "provider":  "dashscope",
        "context":   32_000,
        "vision":    False,
        "free":      False,
    },
    "qwen-vl-max": {
        "id":        "qwen-vl-max",
        "provider":  "dashscope",
        "context":   32_000,
        "vision":    True,
        "free":      False,
    },
    # Ollama (offline)
    "phi3:mini": {
        "id":        "phi3:mini",
        "provider":  "ollama",
        "context":   4_096,
        "free":      True,
        "offline":   True,
    },
}

def resolve(model_alias: str) -> dict:
    """Resolve a short alias to full model metadata."""
    return REGISTRY.get(model_alias) or _fetch_from_models_dev(model_alias)

def _fetch_from_models_dev(alias: str) -> dict:
    """Live fallback: query models.dev public API."""
    try:
        data = httpx.get(MODELS_DEV_URL, timeout=5).json()
        return next((m for m in data if alias in (m.get("id",""), m.get("name",""))), {})
    except Exception:
        return {}
```


## 14. 🔌 Providers & Connectors

Modelled after OpenCode's provider system — each provider is a self-contained class with a standard `chat()` interface. Drop a new file in `brain/providers/` to add a provider.

### Provider interface (`brain/providers/base.py`)
```python
# brain/providers/base.py
from abc import ABC, abstractmethod

class BaseProvider(ABC):
    name    : str = ""
    base_url: str = ""

    @abstractmethod
    def chat(self, messages: list, model: str, **kwargs) -> str: ...

    @abstractmethod
    def models(self) -> list[str]:
        """Return list of available model IDs from this provider."""
        ...
```

### OpenRouter (`brain/providers/openrouter.py`)
```python
import httpx, os
from brain.providers.base import BaseProvider

class OpenRouterProvider(BaseProvider):
    name     = "openrouter"
    base_url = "https://openrouter.ai/api/v1"

    def __init__(self, model: str = "google/gemini-2.0-flash-exp:free"):
        self.model   = model
        self.api_key = os.getenv("OPENROUTER_API_KEY")

    def chat(self, messages, model=None, **kwargs) -> str:
        resp = httpx.post(f"{self.base_url}/chat/completions",
            headers={"Authorization": f"Bearer {self.api_key}"},
            json={"model": model or self.model, "messages": messages},
            timeout=120)
        resp.raise_for_status()
        return resp.json()["choices"][0]["message"]["content"]

    def models(self) -> list[str]:
        resp = httpx.get(f"{self.base_url}/models",
            headers={"Authorization": f"Bearer {self.api_key}"})
        return [m["id"] for m in resp.json().get("data", [])]
```
